  geom_smooth(aes(x = TP, y = (yeasts/362), color = Treatment1))+
  scale_color_manual(values = c("black", "blue", "red"))+
  theme_bw()+
  labs(colour="Drought treatment")+
  theme(legend.title = element_text(colour="black", size=15, face="bold"),
